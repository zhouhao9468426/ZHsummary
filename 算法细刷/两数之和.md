### 1.两数之和  
>https://leetcode-cn.com/problems/two-sum/   
***  
暴力解法：
```
class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int,int> umap;
        int len = nums.size();
        for(int i=0; i<len; ++i)
        {
            auto it = umap.find(target-nums[i]);
            if(it != umap.end())
            {
                return {it->second, i};
            }
            umap[nums[i]] = i;
        }
        return {};
    }
};
```
